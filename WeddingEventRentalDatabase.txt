DROP TABLE CUSTOMER CASCADE CONSTRAINTS;
CREATE TABLE CUSTOMER (
CID INTEGER NOT NULL,
CFNAME VARCHAR2(30),
CLNAME VARCHAR2(30),
CUSTOMERSTREET VARCHAR2(30),
CUSTOMERCITY VARCHAR2(30),
CUSTOMERSTATE CHAR(2),
CUSTOMERZIP INTEGER,
CUSTOMERPHONE CHAR(12),
PRIMARY KEY (CID)
);
DROP TABLE ORDERS CASCADE CONSTRAINTS;
CREATE TABLE ORDERS (
ORDERID INTEGER NOT NULL,
ORDERDATE VARCHAR2(40),
ORDERTIME VARCHAR2(12),
ORDERSTREET VARCHAR2(30),
ORDERCITY VARCHAR2(30),
ORDERSTATE CHAR(2),
ORDERZIP INTEGER,
CID INTEGER,
PRIMARY KEY (ORDERID),
FOREIGN KEY (CID) REFERENCES CUSTOMER (CID)
);
DROP TABLE WAREHOUSE CASCADE CONSTRAINTS;
CREATE TABLE WAREHOUSE (
WAREHOUSEID INTEGER,
WAREHOUSESTREET VARCHAR2(40),
WAREHOUSECITY VARCHAR2(40),
WAREHOUSESTATE CHAR(2),
WAREHOUSEZIP INTEGER,
PRIMARY KEY (WAREHOUSEID)
);
DROP TABLE EQUIPMENT CASCADE CONSTRAINTS;
CREATE TABLE EQUIPMENT (
EQUIPMENTID INTEGER,
EQUIPMENTTYPE VARCHAR2(30),
EQUIPMENTQUANTITY VARCHAR2(30),
EQUIPMENTCOLOR VARCHAR2(30),
EQUIPMENTDESCRIPTION VARCHAR2(30),
WAREHOUSEID INTEGER,
PRIMARY KEY (EQUIPMENTID),
FOREIGN KEY (WAREHOUSEID) REFERENCES WAREHOUSE (WAREHOUSEID)
);
DROP TABLE PAYMENT CASCADE CONSTRAINTS;
CREATE TABLE PAYMENT(
PAYMENTID INTEGER NOT NULL,
PAYMENTAMOUNT DECIMAL(20) NOT NULL,
PAYMENTTERMS VARCHAR2(30),
CID INTEGER,
PRIMARY KEY (PAYMENTID),
FOREIGN KEY (CID) REFERENCES CUSTOMER (CID));
DROP TABLE EVENT CASCADE CONSTRAINTS;
CREATE TABLE EVENT (
EID INTEGER,
MYEVENT VARCHAR2(40),
OEQUANTITY INTEGER,
ORDERID INTEGER,
EQUIPMENTID INTEGER,
PRIMARY KEY (EID),
FOREIGN KEY (ORDERID) REFERENCES ORDERS (ORDERID),
FOREIGN KEY (EQUIPMENTID) REFERENCES EQUIPMENT (EQUIPMENTID));
DROP TABLE AMOUNTS CASCADE CONSTRAINTS;
CREATE TABLE AMOUNTS(
AMTID INTEGER,
OPDATE VARCHAR2(40),
OPTIME VARCHAR2(40),
PAYMENTID INTEGER,
ORDERID INTEGER,
PRIMARY KEY (AMTID),
FOREIGN KEY (PAYMENTID) REFERENCES PAYMENT (PAYMENTID),
FOREIGN KEY (ORDERID) REFERENCES ORDERS (ORDERID)
);
